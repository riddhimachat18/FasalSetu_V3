 SETUP ORDER (first run)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1.  cp .env.example .env                         # add ANTHROPIC_API_KEY
2.  pip install -r requirements.txt
3.  python scripts/train_npk_model.py \
        --data npk_iot_processed.csv             # → populates models/
4.  python scripts/ingest_schemes.py --verify    # → populates data/chroma_db/
5.  # Open scripts/train_disease_model_colab.py in Colab → Run All (~35 min)
    # Download disease_model.pth + disease_labels.json → models/
6.  uvicorn main:app --reload --port 8000        # start API
7.  open http://localhost:8000                   # browser demo UI
8.  python scripts/demo_cli.py --scenario compliance   # showstopper demo
